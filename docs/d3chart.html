    getXScale: () ->
      d = @_data.map(@barsDataAccessor)
      min_x = d3.min(d, (d) -> d3.min(d, (d) -> d.x))
      max_x = d3.max(d, (d) -> d3.max(d, (d) -> d.x))
      return d3.scale.ordinal()
          .domain(d3.range(min_x, max_x + 1))
          .rangeRoundBands([0, @options.plotBox.width], 0.1, 0.1)
